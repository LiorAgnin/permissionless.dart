@override
Future<String> signTypedData({
  required BigInt chainId,
  required EthAddress verifyingContract,
  required String hashedMessage,
}) async {
  // Sign the EIP-712 typed data for LightAccountMessage
  final typedData = TypedData(
    domain: TypedDataDomain(
      name: 'LightAccount',
      version: '1',
      chainId: chainId,
      verifyingContract: verifyingContract,
    ),
    types: {
      'LightAccountMessage': [
        const TypedDataField(name: 'message', type: 'bytes'),
      ],
    },
    primaryType: 'LightAccountMessage',
    message: {'message': hashedMessage},
  );

  final hash = hashTypedData(typedData);
  return signHash(hash);
}
