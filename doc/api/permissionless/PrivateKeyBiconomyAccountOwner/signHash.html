@override
Future<String> signHash(String messageHash) async {
  // Biconomy uses eth_sign format (with Ethereum Signed Message prefix)
  // This matches viem's signMessage({ raw: hash }) behavior
  final hashBytes = Hex.decode(messageHash);

  // Add Ethereum Signed Message prefix: "\x19Ethereum Signed Message:\n32"
  final prefix = '\x19Ethereum Signed Message:\n${hashBytes.length}';
  final prefixBytes = Uint8List.fromList(prefix.codeUnits);
  final prefixedMessage = Uint8List(prefixBytes.length + hashBytes.length)
    ..setRange(0, prefixBytes.length, prefixBytes)
    ..setRange(
      prefixBytes.length,
      prefixBytes.length + hashBytes.length,
      hashBytes,
    );

  // Hash the prefixed message and sign
  final prefixedHash = keccak256(prefixedMessage);
  final sig = _privateKey.signToEcSignature(prefixedHash);

  // Encode as r (32 bytes) + s (32 bytes) + v (1 byte)
  final r = Hex.fromBigInt(sig.r, byteLength: 32);
  final s = Hex.fromBigInt(sig.s, byteLength: 32);
  // Biconomy may need V adjustment (add 27 if v < 27)
  var v = sig.v;
  if (v < 27) v += 27;
  final vHex = Hex.fromBigInt(BigInt.from(v), byteLength: 1);

  return Hex.concat([r, s, vHex]);
}
